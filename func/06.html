Identity 代码 #

js
const Identity = x => ({
  map: f => Identity(f(x)),
  valueOf: () => x,
  inspect: () => `Identity {${x}}`
})

// 值为 21
const computeBox = Identity(10)
                    .map(add4)
                    .map(multiply3)
                    .map(divide2)
                    .valueOf();

Maybe Functor：识别空数据 #

Maybe Functor 代码 #

js
const isEmpty = x => x === undefined || x === null  

const Maybe = x => ({
  map: f => isEmpty(x) ? Maybe(null) : Maybe(f(x)),  
  valueOf: () => x,  
  inspect: () => `Maybe {${x}}`
})
